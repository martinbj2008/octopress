TARGETS =    111.cpp 222.cpp 333.cpp
OTARGETS= $(subst cpp,o,$(TARGETS))
LTARGETS= $(subst cpp,lo,$(TARGETS))
%.o: %.cpp
	++    -c -o $@    $<
all: objs libs
